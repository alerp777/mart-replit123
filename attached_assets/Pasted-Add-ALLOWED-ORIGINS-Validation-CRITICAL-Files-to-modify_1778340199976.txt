What's broken: If ALLOWED_ORIGINS env var is not set, the CORS policy may allow ALL origins. This allows any website to make requests to your API on behalf of your users. In production, this is a CRITICAL vulnerability.

What to do:

Add strict validation of ALLOWED_ORIGINS at app startup
If NODE_ENV === 'production' AND ALLOWED_ORIGINS is empty/missing:
Throw error and REFUSE TO BOOT
Log error and exit with code 1
If NODE_ENV === 'development' AND ALLOWED_ORIGINS is empty:
Log a WARNING
Allow only localhost and 127.0.0.1
Parse ALLOWED_ORIGINS as comma-separated list
On every request, STRICTLY check Origin header against whitelist

// app.ts
function validateCORS() {
  const allowedOrigins = process.env.ALLOWED_ORIGINS;
  
  if (process.env.NODE_ENV === 'production') {
    if (!allowedOrigins || allowedOrigins.trim() === '') {
      logger.error(
        '[CRITICAL] ALLOWED_ORIGINS not set in production. App will not start.'
      );
      process.exit(1);
    }
  } else if (process.env.NODE_ENV === 'development') {
    if (!allowedOrigins || allowedOrigins.trim() === '') {
      logger.warn(
        '[WARNING] ALLOWED_ORIGINS not set. Defaulting to localhost only.'
      );
      return ['http://localhost:5173', 'http://localhost:3001', 'http://localhost:3002', 'http://127.0.0.1:5000'];
    }
  }
  
  return allowedOrigins.split(',').map(o => o.trim());
}

const corsOrigins = validateCORS();

app.use(cors({
  origin: (origin, callback) => {
    if (!origin || corsOrigins.includes(origin)) {
      callback(null, true);
    } else {
      logger.warn('[SECURITY:CORS] Blocked request from unauthorized origin', { origin });
      callback(new Error('CORS policy violation'));
    }
  },
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH', 'OPTIONS'],
  allowedHeaders: ['Content-Type', 'Authorization', 'X-Request-ID'],
  maxAge: 3600
}));

logger.info('[STARTUP] CORS enabled for origins:', corsOrigins);
